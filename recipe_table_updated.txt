create table recipe (
    recipe_id serial primary key.
    recipe_name varchar(25),
    cooking_time varchar(10),
    calories varchar(20),
    fat varchar (20),
    protein varchar (20),
    salt varchar (20),
    sugar varchar (20),
    fibre varchar (20), 
    carbonhydrates varchar(20)
);
